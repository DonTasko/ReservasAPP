/** * DON TASKO v6.2 - RESERVAS COMPLETO * Com Email e Google Calendar */ // ========== CONFIGURAÇÕES ========== const CONFIG_GERAL = { CALENDAR_ID: 'primary', // ou coloque o ID do seu calendário específico DURACAO_RESERVA_MINUTOS: 90, // 1 hora e 30 minutos EMAIL_ADMIN: 'dontasko.geral@gmail.com' }; // ========== doGet - JSONP ========== function doGet(e) { // Se não houver parâmetros, retorna HTML simples if (!e || !e.parameter) { return HtmlService.createHtmlOutput("

Don Tasko - API de Reservas

API está funcionando.

"); } // Obter ação const action = e.parameter.action; const callback = e.parameter.callback; // Se não houver ação, retorna erro if (!action) { const resultado = { status: "error", message: "Ação não especificada" }; if (callback) { return ContentService .createTextOutput(callback + '(' + JSON.stringify(resultado) + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService .createTextOutput(JSON.stringify(resultado)) .setMimeType(ContentService.MimeType.JSON); } // Processar ação let resultado; try { switch(action) { case 'getDiasFechados': resultado = getDiasFechados(); break; case 'getHorarios': resultado = getHorarios(); break; case 'reservaSite': resultado = processarReserva(e.parameter); break; default: resultado = { status: "error", message: "Ação desconhecida: " + action }; } } catch(error) { resultado = { status: "error", message: error.toString() }; } // Retornar JSONP ou JSON if (callback) { return ContentService .createTextOutput(callback + '(' + JSON.stringify(resultado) + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } else { return ContentService .createTextOutput(JSON.stringify(resultado)) .setMimeType(ContentService.MimeType.JSON); } } // ========== DIAS FECHADOS ========== function getDiasFechados() { try { // Tentar ler da sheet Calendario_Fecho const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName('Calendario_Fecho'); if (sheet) { const dados = sheet.getDataRange().getValues(); const diasFechados = []; for (let i = 0; i < dados.length; i++) { const valor = dados[i][0]; if (valor) { if (valor instanceof Date) { const diaSemana = valor.toLocaleDateString('pt-PT', { weekday: 'long' }).toLowerCase(); diasFechados.push(diaSemana); const dataFormatada = Utilities.formatDate(valor, Session.getScriptTimeZone(), 'dd/MM/yyyy'); diasFechados.push(dataFormatada); } else { diasFechados.push(valor.toString().toLowerCase().trim()); } } } if (diasFechados.length > 0) { return [...new Set(diasFechados)]; } } // Valores padrão: Segunda, Terça, Quarta return ['segunda', 'terça', 'quarta']; } catch(error) { console.error('Erro getDiasFechados:', error); return ['segunda', 'terça', 'quarta']; } } // ========== HORÁRIOS ========== function getHorarios() { return { almoco: '12:00-14:15', jantar: '19:30-20:15' }; } // ========== PROCESSAR RESERVA ========== function processarReserva(params) { try { // Validar dados obrigatórios if (!params.nome || params.nome.trim() === '') { return { status: "error", message: "Nome é obrigatório" }; } if (!params.email || !params.email.includes('@')) { return { status: "error", message: "Email inválido" }; } const telemovel = params.telemovel ? params.telemovel.toString().replace(/\D/g, '') : ''; if (telemovel.length < 9) { return { status: "error", message: "Telemóvel inválido (mínimo 9 dígitos)" }; } if (!params.data) { return { status: "error", message: "Data é obrigatória" }; } const pax = parseInt(params.pax); if (isNaN(pax) || pax < 1) { return { status: "error", message: "Número de pessoas inválido" }; } if (!params.horario) { return { status: "error", message: "Horário é obrigatório" }; } // Verificar se a data está fechada const diasFechados = getDiasFechados(); const dataObj = new Date(params.data); const diaSemana = dataObj.toLocaleDateString('pt-PT', { weekday: 'long' }).toLowerCase(); const dia = dataObj.getDate().toString().padStart(2, '0'); const mes = (dataObj.getMonth() + 1).toString().padStart(2, '0'); const ano = dataObj.getFullYear(); const dataFormatada = `${dia}/${mes}/${ano}`; if (diasFechados.includes(diaSemana) || diasFechados.includes(params.data) || diasFechados.includes(dataFormatada)) { return { status: "error", message: "O restaurante está fechado nesta data. Estamos abertos de Quinta a Domingo." }; } // Verificar se a data é anterior a hoje const hoje = new Date(); hoje.setHours(0, 0, 0, 0); if (dataObj < hoje) { return { status: "error", message: "Não é possível fazer reservas para datas passadas" }; } // Salvar na sheet Reservas const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName('Reservas'); if (!sheet) { sheet = ss.insertSheet('Reservas'); sheet.appendRow(['ID', 'DATA', 'HORARIO', 'NOME', 'PAX', 'EMAIL', 'TELEMOVEL', 'OBS', 'STATUS', 'DATA_REGISTO']); } const id = 'RES' + Date.now(); sheet.appendRow([ id, params.data, params.horario, params.nome, pax, params.email, telemovel, params.obs || '', 'Confirmada', new Date() ]); // Criar evento no Google Calendar let eventoCalendar = null; try { eventoCalendar = criarEventoCalendar(params, id); } catch(e) { console.error('Erro ao criar evento no calendário:', e); } // Enviar email de confirmação para o cliente let emailEnviado = false; let emailErro = null; try { emailEnviado = enviarEmailConfirmacao(params, id); } catch(e) { emailErro = e.toString(); console.error('Erro ao enviar email:', e); } // Enviar email para o admin (opcional) try { enviarEmailAdmin(params, id); } catch(e) { console.error('Erro ao enviar email para admin:', e); } return { status: "success", message: "Reserva confirmada com sucesso! Enviamos um email de confirmação para " + params.email, id: id, emailEnviado: emailEnviado, emailErro: emailErro, calendarEvento: eventoCalendar ? "Criado" : "Falhou" }; } catch(error) { console.error('Erro processarReserva:', error); return { status: "error", message: "Erro interno: " + error.toString() }; } } // ========== CRIAR EVENTO NO GOOGLE CALENDAR ========== function criarEventoCalendar(dados, id) { try { // Parse da data e hora const [ano, mes, dia] = dados.data.split('-'); const [hora, minuto] = dados.horario.split(':'); const dataInicio = new Date(ano, mes - 1, dia, hora, minuto); const dataFim = new Date(dataInicio.getTime() + (CONFIG_GERAL.DURACAO_RESERVA_MINUTOS * 60 * 1000)); // Obter calendário let calendar; try { calendar = CalendarApp.getCalendarById(CONFIG_GERAL.CALENDAR_ID); if (!calendar) { calendar = CalendarApp.getDefaultCalendar(); } } catch(e) { calendar = CalendarApp.getDefaultCalendar(); } // Criar evento const evento = calendar.createEvent( `🍽️ ${dados.pax}pax - ${dados.nome}`, dataInicio, dataFim, { description: `ID: ${id}\n` + `Email: ${dados.email}\n` + `Telemóvel: ${dados.telemovel}\n` + `Pessoas: ${dados.pax}\n` + `Observações: ${dados.obs || 'Nenhuma'}\n` + `\n---\nReserva feita pelo site`, location: "Don Tasko Restaurante, Castelo Branco", guests: dados.email, sendInvites: false } ); console.log(`✅ Evento criado no calendário: ${evento.getId()}`); return evento.getId(); } catch(error) { console.error('❌ Erro ao criar evento no calendário:', error); return null; } } // ========== ENVIAR EMAIL DE CONFIRMAÇÃO PARA O CLIENTE ========== function enviarEmailConfirmacao(dados, id) { try { if (!dados.email) { console.log('❌ Email não fornecido'); return false; } // Formatar data para português const dataObj = new Date(dados.data); const dataFormatada = dataObj.toLocaleDateString('pt-PT', { weekday: 'long', day: 'numeric', month: 'long', year: 'numeric' }); const assunto = "🍽️ Don Tasko - Confirmação da sua Reserva"; const corpoHTML = `
